local function _skipFont(obj)
    if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
        -- не применять шрифт к топбару
        if obj:IsDescendantOf(TopbarGui and TopbarGui or game) and TopbarGui then return true end
        if obj.TextStrokeTransparency ~= 1 then return true end
        local cur = tostring(obj.FontFace)
        return cur == _BUILDER_FONT or string.find(cur, "BuilderIcons")
    end
    return true
end
local function _applyFont(obj)
    if (obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox")) and not _skipFont(obj) then
        obj.FontFace = _SENKO_FONT
    end
end

PRED_TIME=0.09 PRED_HISTORY=20 predHistory={}
local function recordPredHistory(userId, pos, vel)
    if not predHistory[userId] then predHistory[userId]={} end
    local h=predHistory[userId]
    table.insert(h,{pos=pos, vel=vel, t=tick()})
    while #h>PRED_HISTORY do table.remove(h,1) end
end
local function getPredictedPos(targetHRP, userId)
    local rawVel = targetHRP.AssemblyLinearVelocity
    recordPredHistory(userId, targetHRP.Position, rawVel)
    local h = predHistory[userId]
    if not h or #h < 3 then return targetHRP.Position + rawVel * PRED_TIME * 1.5 end
    local totalWeight, weightedVel = 0, Vector3.new(0,0,0)
    for i, sample in ipairs(h) do
        local w = (i / #h) ^ 2.5
        weightedVel = weightedVel + sample.vel * w
        totalWeight = totalWeight + w
    end
    if totalWeight > 0 then weightedVel = weightedVel / totalWeight end
    local newestVel = h[#h].vel
    local oldIdx = math.max(1, #h - 6)
    local dt_accel = h[#h].t - h[oldIdx].t
    local accel = dt_accel > 0.001 and (newestVel - h[oldIdx].vel) / dt_accel or Vector3.new(0,0,0)
    local blendXZ = Vector3.new(weightedVel.X*0.25+newestVel.X*0.75, 0, weightedVel.Z*0.25+newestVel.Z*0.75)
    local horizMag = Vector3.new(blendXZ.X, 0, blendXZ.Z).Magnitude
    local extraLead = math.clamp(horizMag / 12, 0, 3.5)
    local leadTime = PRED_TIME * (2.4 + extraLead)
    local velY = newestVel.Y
    local vertLead = math.abs(velY) > 4 and (newestVel.Y*0.7 + accel.Y*PRED_TIME*0.5) or newestVel.Y*0.2
    return targetHRP.Position
        + Vector3.new(blendXZ.X, vertLead, blendXZ.Z) * leadTime
        + Vector3.new(accel.X, 0, accel.Z) * (leadTime * leadTime * 0.4)
end

local function tpToFloor()
    local h = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    if not h then return end
    local origin = h.Position
    local rp = RaycastParams.new()
    rp.FilterDescendantsInstances = {player.Character}
    rp.FilterType = Enum.RaycastFilterType.Exclude
    local result = workspace:Raycast(origin + Vector3.new(0,5,0), Vector3.new(0,-1000,0), rp)
    if result then
        local hum = player.Character:FindFirstChildOfClass("Humanoid")
        local hip = hum and hum.HipHeight or 2.3
        h.CFrame = CFrame.new(origin.X, result.Position.Y + hip + 0.1, origin.Z)
        h.AssemblyLinearVelocity = Vector3.new(h.AssemblyLinearVelocity.X, 0, h.AssemblyLinearVelocity.Z)
    end
end

end

-- ── NO CAMERA COLLISION ───────────────────────────────────────────────────────
local function enableNoCamCollision()
    noCamCollisionEnabled=true
    if noCamCollisionConn then noCamCollisionConn:Disconnect() end
    noCamCollisionConn = RunService.RenderStepped:Connect(function()
        if not noCamCollisionEnabled then return end
        local ch=player.Character; if not ch then return end
        local cam=workspace.CurrentCamera; if not cam then return end
        local hrp=ch:FindFirstChild("HumanoidRootPart"); if not hrp then return end
        local camPos  = cam.CFrame.Position
        local charPos = hrp.Position + Vector3.new(0,1.5,0)
        local toChar  = charPos - camPos
        local params  = RaycastParams.new()
        params.FilterType = Enum.RaycastFilterType.Exclude
        params.FilterDescendantsInstances = {ch}
        params.IgnoreWater = true
        local hit={}; local origin=camPos; local remaining=toChar
        for _=1,12 do
            if remaining.Magnitude < 0.2 then break end
            local res=workspace:Raycast(origin, remaining, params)
            if not res then break end
            local p=res.Instance
            if p and p:IsA("BasePart") and not p:IsDescendantOf(ch) then
                hit[p]=true
                if noCamParts[p]==nil then noCamParts[p]=p.LocalTransparencyModifier end
                p.LocalTransparencyModifier=1
            end
            origin=res.Position+remaining.Unit*0.02; remaining=charPos-origin
        end
        for p,orig in pairs(noCamParts) do
            if not hit[p] then
                pcall(function() if p and p.Parent then p.LocalTransparencyModifier=orig end end)
                noCamParts[p]=nil
            end
        end
    end)
end
local function disableNoCamCollision()
    noCamCollisionEnabled=false
    if noCamCollisionConn then noCamCollisionConn:Disconnect(); noCamCollisionConn=nil end
    for p,orig in pairs(noCamParts) do
        pcall(function() if p and p.Parent then p.LocalTransparencyModifier=orig end end)
    end

-- ── ANTI RAGDOLL ─────────────────────────────────────────────────────────────
local function startAntiRagdoll()
    if #AntiRagdollConns>0 then return end
    local c=player.Character or player.CharacterAdded:Wait()
    local humanoid=c:WaitForChild("Humanoid")
    local root=c:WaitForChild("HumanoidRootPart")
    local animator=humanoid:WaitForChild("Animator")
    local maxVelocity=40; local clampVelocity=25; local maxClamp=15
    local lastVelocity=Vector3.new(0,0,0)
    local function IsRagdollState()
        local st=humanoid:GetState()
        return st==Enum.HumanoidStateType.Physics or st==Enum.HumanoidStateType.Ragdoll
            or st==Enum.HumanoidStateType.FallingDown or st==Enum.HumanoidStateType.GettingUp
    end
    local function CleanRagdollEffects()
        for _,obj in pairs(c:GetDescendants()) do
            if obj:IsA("BallSocketConstraint") or obj:IsA("NoCollisionConstraint") or obj:IsA("HingeConstraint")
                or (obj:IsA("Attachment") and (obj.Name=="A" or obj.Name=="B")) then obj:Destroy()
            elseif obj:IsA("BodyVelocity") or obj:IsA("BodyPosition") or obj:IsA("BodyGyro") then obj:Destroy()
            elseif obj:IsA("Motor6D") then obj.Enabled=true end
        end
        for _,track in pairs(animator:GetPlayingAnimationTracks()) do
            local an=track.Animation and track.Animation.Name:lower() or ""
            if an:find("rag") or an:find("fall") or an:find("hurt") or an:find("down") then track:Stop(0) end
        end
    end
    local function ReEnableControls()
        pcall(function() require(player:WaitForChild("PlayerScripts"):WaitForChild("PlayerModule")):GetControls():Enable() end)
    end
    table.insert(AntiRagdollConns,humanoid.StateChanged:Connect(function()
        if IsRagdollState() then humanoid:ChangeState(Enum.HumanoidStateType.Running) CleanRagdollEffects() workspace.CurrentCamera.CameraSubject=humanoid ReEnableControls() end
    end))
    table.insert(AntiRagdollConns,RunService.Heartbeat:Connect(function()
        if not antiRagdollEnabled then return end
        if IsRagdollState() then
            CleanRagdollEffects()
            local vel=root.AssemblyLinearVelocity
            if (vel-lastVelocity).Magnitude>maxVelocity and vel.Magnitude>clampVelocity then
                root.AssemblyLinearVelocity=vel.Unit*math.min(vel.Magnitude,maxClamp) end
            lastVelocity=vel
        end
    end))
    table.insert(AntiRagdollConns,c.DescendantAdded:Connect(function() if IsRagdollState() then CleanRagdollEffects() end end))
    table.insert(AntiRagdollConns,player.CharacterAdded:Connect(function(newChar)
        c=newChar humanoid=newChar:WaitForChild("Humanoid") root=newChar:WaitForChild("HumanoidRootPart")
        animator=humanoid:WaitForChild("Animator") lastVelocity=Vector3.new(0,0,0)
        ReEnableControls() CleanRagdollEffects()
    end))
    ReEnableControls() CleanRagdollEffects()
end
local function stopAntiRagdoll()
    for _,conn in pairs(AntiRagdollConns) do conn:Disconnect() end AntiRagdollConns={}
end

local function isTargetValid(targetChar)
    if not targetChar then return false end
    local hum=targetChar:FindFirstChildOfClass("Humanoid")
    local hrp=targetChar:FindFirstChild("HumanoidRootPart")
    local ff=targetChar:FindFirstChildOfClass("ForceField")
    return hum and hrp and hum.Health>0 and not ff
end
local function getBestTarget(myHRP)
    if lockedTarget and isTargetValid(lockedTarget) then return lockedTarget:FindFirstChild("HumanoidRootPart"),lockedTarget end
    local shortestDist=math.huge local newTargetChar=nil local newTargetHRP=nil
    for _,targetPlayer in ipairs(Players:GetPlayers()) do
        if targetPlayer~=player and isTargetValid(targetPlayer.Character) then
            local targetHRP=targetPlayer.Character:FindFirstChild("HumanoidRootPart")
            local distance=(targetHRP.Position-myHRP.Position).Magnitude
            if distance<shortestDist then shortestDist=distance newTargetHRP=targetHRP newTargetChar=targetPlayer.Character end
        end
    end
    lockedTarget=newTargetChar return newTargetHRP,newTargetChar
end

                local speed=targetVelocity.Magnitude
                local dynTime=math.clamp(speed/150,0.05,0.2)
                predictedPos=targetHRP.Position+(targetVelocity*dynTime)
            end
            local dirToTarget=(predictedPos-currentHRP.Position)
            local distance3D=dirToTarget.Magnitude
            local targetStandPos=predictedPos
            if distance3D>0 then targetStandPos=predictedPos-(dirToTarget.Unit*(MELEE_OFFSET*0.5)) end
            align.CFrame=CFrame.lookAt(currentHRP.Position,predictedPos)
            local moveDir=(targetStandPos-currentHRP.Position)
            local distToStandPos=moveDir.Magnitude
            if distToStandPos>1 then
                currentHRP.AssemblyLinearVelocity=Vector3.new(
                    moveDir.Unit.X*AIMBOT_SPEED,
                    targetHRP.AssemblyLinearVelocity.Y,
                    moveDir.Unit.Z*AIMBOT_SPEED
                )
            else
                currentHRP.AssemblyLinearVelocity=Vector3.new(
                    targetVelocity.X,
                    targetHRP.AssemblyLinearVelocity.Y,
                    targetVelocity.Z
                )
            end
            if lockPlayerMode=="auto" and distToStandPos<=BAT_ENGAGE_RANGE then
                if bat and bat.Parent==myChar then pcall(function() bat:Activate() end) end
            end
        else
            lockedTarget=nil currentHRP.AssemblyLinearVelocity=Vector3.new(0,currentHRP.AssemblyLinearVelocity.Y,0) aimbotHighlight.Adornee=nil
            if predDot and predDot.Parent then predDot.Transparency=1 end
        end
    end)
end

    local c=player.Character local h=c and c:FindFirstChild("HumanoidRootPart") local hum=c and c:FindFirstChildOfClass("Humanoid")

end

local function detectMyBaseSide()
    local plots=workspace:FindFirstChild("Plots") if not plots then return nil end
    for _,plot in ipairs(plots:GetChildren()) do

local apDispToggleCb = nil
local function stopAutoPlay()
    autoPlayOn=false
    if autoPlayConn then autoPlayConn:Disconnect() autoPlayConn=nil end
    autoPlayPhase=1 autoPlaySide=nil
    local hh=getHum() if hh then hh:Move(Vector3.zero,false) end
    if apDispToggleCb then pcall(apDispToggleCb, false) end
end
local function startAutoPlay()
    stopAutoPlay()
    local mySide=detectMyBaseSide()
    autoPlaySide=mySide=="Left" and "Right" or "Left"
    local side=autoPlaySide
    local p1,p2,pFinal=getAPWaypoints(side)
    autoPlayOn=true autoPlayPhase=1
    if autoPlayConn then autoPlayConn:Disconnect() end
    autoPlayConn=RunService.Heartbeat:Connect(function()
        if not autoPlayOn or not gHrp or not gHum then return end
        local nspd,sspd=getSBSpeeds()
        if not speedBoostEnabled then nspd=16 sspd=16 end
        local sdelay=AP_STEAL_DELAY
        p1,p2,pFinal=getAPWaypoints(side)
        if autoPlayPhase==1 then
            local d=Vector3.new(p1.X-gHrp.Position.X,0,p1.Z-gHrp.Position.Z)
            if d.Magnitude<1 then autoPlayPhase=2 return end
            local md=d.Unit gHum:Move(md,false)
            gHrp.AssemblyLinearVelocity=Vector3.new(md.X*nspd,gHrp.AssemblyLinearVelocity.Y,md.Z*nspd)
        elseif autoPlayPhase==2 then
            local d=Vector3.new(p2.X-gHrp.Position.X,0,p2.Z-gHrp.Position.Z)
            if d.Magnitude<1 then
                autoPlayPhase=0 gHum:Move(Vector3.zero,false) gHrp.AssemblyLinearVelocity=Vector3.zero
                task.delay(sdelay,function() if autoPlayOn then autoPlayPhase=3 end end) return
            end
            local md=d.Unit gHum:Move(md,false)
            gHrp.AssemblyLinearVelocity=Vector3.new(md.X*nspd,gHrp.AssemblyLinearVelocity.Y,md.Z*nspd)
        elseif autoPlayPhase==0 then return
        elseif autoPlayPhase==3 then
            local d=Vector3.new(p1.X-gHrp.Position.X,0,p1.Z-gHrp.Position.Z)
            if d.Magnitude<1 then autoPlayPhase=4 return end
            local md=d.Unit gHum:Move(md,false)
            gHrp.AssemblyLinearVelocity=Vector3.new(md.X*sspd,gHrp.AssemblyLinearVelocity.Y,md.Z*sspd)
        elseif autoPlayPhase==4 then
            local d=Vector3.new(pFinal.X-gHrp.Position.X,0,pFinal.Z-gHrp.Position.Z)
            if d.Magnitude<1 then
                gHum:Move(Vector3.zero,false) gHrp.AssemblyLinearVelocity=Vector3.zero
                stopAutoPlay() return
            end
            local md=d.Unit gHum:Move(md,false)
            gHrp.AssemblyLinearVelocity=Vector3.new(md.X*sspd,gHrp.AssemblyLinearVelocity.Y,md.Z*sspd)
        end
    end)
end

local function findPromptForAnimal(ad)
    if not ad then return nil end
    local cp=promptCache[ad.uid] if cp and cp.Parent then return cp end

    for _,p in ipairs(att:GetChildren()) do if p:IsA("ProximityPrompt") then promptCache[ad.uid]=p return p end end
end
local function buildCallbacks(prompt)
    if stealCache[prompt] then return end
    local data={hold={},trigger={},ready=true}
    pcall(function()
        if getconnections then
            for _,c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do if type(c.Function)=="function" then table.insert(data.hold,c.Function) end end
            for _,c in ipairs(getconnections(prompt.Triggered)) do if type(c.Function)=="function" then table.insert(data.trigger,c.Function) end end
        end
    end)
    if #data.hold>0 or #data.trigger>0 then stealCache[prompt]=data end
end
local function nearestAnimal()
    local h=getHRP() if not h then return nil end
    local best,bestD=nil,math.huge
    for _,ad in ipairs(animalCache) do
        if not isMyBase(ad.plot) and ad.worldPosition then
            local d=(h.Position-ad.worldPosition).Magnitude if d<bestD then bestD=d best=ad end
        end
    end
    return best
end
local function execInstantGrab(prompt)
    local data=stealCache[prompt] if not data or not data.ready then return false end
    data.ready=false instantGrabStealing=true isStealing=true
    task.spawn(function()
        local grabbed=false
        if #data.hold>0 or #data.trigger>0 then
            pcall(function()
                for _,fn in ipairs(data.hold) do task.spawn(fn) end
                task.wait(0.005)
                for _,fn in ipairs(data.trigger) do task.spawn(fn) end
                grabbed=true
            end)
        end
        if not grabbed then
            pcall(function()
                if fireproximityprompt then fireproximityprompt(prompt); grabbed=true end
            end)
        end
        if not grabbed then
            pcall(function()
                prompt.MaxActivationDistance=9999
                prompt:InputHoldBegin(); task.wait(STEAL_DURATION); prompt:InputHoldEnd()
            end)
        end
        task.wait(STEAL_DELAY) data.ready=true instantGrabStealing=false isStealing=false
    end)
    return true
end
local function startInstantGrab()
    if instantGrabConn then return end
    instantGrabConn=RunService.Heartbeat:Connect(function()
        if not instantGrabEnabled then return end
        if instantGrabStealing then return end
        local target=nearestAnimal() if not target then return end
        local h=getHRP() if not h then return end
        if (h.Position-target.worldPosition).Magnitude>STEAL_RADIUS then return end
        local prompt=promptCache[target.uid]
        if not prompt or not prompt.Parent then prompt=findPromptForAnimal(target) end
        if prompt then
            buildCallbacks(prompt)
            if stealCache[prompt] then execInstantGrab(prompt)
            else task.spawn(function()
                pcall(function() prompt.MaxActivationDistance=9999 end)
                isStealing=true instantGrabStealing=true
                local grabbed=false
                pcall(function() if fireproximityprompt then fireproximityprompt(prompt); grabbed=true end end)
                if not grabbed then pcall(function() prompt:InputHoldBegin() end); task.wait(STEAL_DURATION); pcall(function() prompt:InputHoldEnd() end) end
                task.wait(STEAL_DELAY); instantGrabStealing=false; isStealing=false
            end) end
        end
    end)
end
local function stopInstantGrab()
    if instantGrabConn then instantGrabConn:Disconnect() instantGrabConn=nil end
    instantGrabStealing=false isStealing=false
end

local function makeDraggable(frame,handle)
    handle=handle or frame
    local dragging,dragStart,startPos=false,nil,nil
    handle.InputBegan:Connect(function(input)
        if lockUIEnabled then return end
        if input.UserInputType==Enum.UserInputType.MouseButton1 or input.UserInputType==Enum.UserInputType.Touch then
            dragging=true; dragStart=input.Position; startPos=frame.Position
            input.Changed:Connect(function() if input.UserInputState==Enum.UserInputState.End then dragging=false end end)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType==Enum.UserInputType.MouseMovement or input.UserInputType==Enum.UserInputType.Touch) then
            local d=input.Position-dragStart
            frame.Position=UDim2.new(startPos.X.Scale,startPos.X.Offset+d.X,startPos.Y.Scale,startPos.Y.Offset+d.Y)
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType==Enum.UserInputType.MouseButton1 or input.UserInputType==Enum.UserInputType.Touch then dragging=false end
    end)
end

            or fps >= 30 and Color3.fromRGB(255, 165, 80)
            or Color3.fromRGB(255, 70, 70)

        fCount = 0
        lTick  = now
    end
end)

        local mainPos=Main.AbsolutePosition; local mainSize=Main.AbsoluteSize
        local centerX=mainPos.X+mainSize.X/2; local centerY=mainPos.Y+mainSize.Y/2
        TweenService:Create(Main,TweenInfo.new(0.15,Enum.EasingStyle.Quad),{Size=UDim2.new(0,newW,0,newH),Position=UDim2.new(0,centerX-newW/2,0,centerY-newH/2)}):Play()
        scaleLbl.Text=math.floor(guiCurrentScale*100).."%"; task.delay(0.16,repositionPanel); task.defer(saveConfig)
    end
    btnMinus.MouseButton1Click:Connect(function() applyScale(guiCurrentScale-SCALE_STEP) end)
    btnPlus.MouseButton1Click:Connect(function() applyScale(guiCurrentScale+SCALE_STEP) end)
    Main:GetPropertyChangedSignal("Visible"):Connect(function()
        if Main.Visible then task.defer(function() repositionPanel(); sizePanel.Visible=true end)
        else sizePanel.Visible=false end
    end)
    RunService.RenderStepped:Connect(function() if not Main.Visible then return end; repositionPanel() end)
    task.defer(function()
        task.wait(0.1)
        local newW=math.floor(BASE_W*guiCurrentScale); local newH=math.floor(BASE_H*guiCurrentScale)
        local mainPos=Main.AbsolutePosition; local mainSize=Main.AbsoluteSize
        local centerX=mainPos.X+mainSize.X/2; local centerY=mainPos.Y+mainSize.Y/2
        Main.Size=UDim2.new(0,newW,0,newH); Main.Position=UDim2.new(0,centerX-newW/2,0,centerY-newH/2)
        scaleLbl.Text=math.floor(guiCurrentScale*100).."%"; task.defer(repositionPanel)
    end)
end

-- PC KEYBINDS
UserInputService.InputBegan:Connect(function(input,processed)
    if processed then return end if changingKeybind then return end
    if input.UserInputType~=Enum.UserInputType.Keyboard then return end
    local k=input.KeyCode

    if Keybinds.AutoPlay and k==Keybinds.AutoPlay then
        local ns=not autoPlayOn
        if ns then if batAimbotEnabled then stopBatAimbot() updateToggle("Lock Player",false) end; startAutoPlay() else stopAutoPlay() end
        updateToggle("Auto Play",ns)
    elseif Keybinds.BatAimbot and k==Keybinds.BatAimbot then
        local ns=not(toggleStates["Lock Player"] and toggleStates["Lock Player"].state)
        if ns then if autoPlayOn then stopAutoPlay() updateToggle("Auto Play",false) end; startBatAimbot() else stopBatAimbot() end
        updateToggle("Lock Player",ns)
    elseif Keybinds.Taunt and k==Keybinds.Taunt then
        if not tauntActive then startTaunt() else stopTaunt() end
    elseif Keybinds.Drop and k==Keybinds.Drop then
        doDrop()
    elseif Keybinds.TpToFloor and k==Keybinds.TpToFloor then
        tpToFloor()
    elseif Keybinds.Float and k==Keybinds.Float then
        floatEnabled=not floatEnabled; if floatEnabled then startFloat() else stopFloat() end; updateToggle("Float",floatEnabled)
    end
end)
